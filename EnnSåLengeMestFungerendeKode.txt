float forgjeAvik = 0.0;
float integral = 0.0;
float deriverte=0.0;

float K_P = 1.2;
float K_I = 0;
float K_D = 0;


float PID2(float avik) {
    integral += avik;
    float derivative = avik - forgjeAvik;
    float output = K_P * avik + K_I * integral + K_D * derivative;
    forgjeAvik = avik;
    if (output >=250){
        output=249;
        integral-=avik;
    }
    else if(output <=-250){
        output=-249;
        integral+=avik;
    }
    return output;
}
